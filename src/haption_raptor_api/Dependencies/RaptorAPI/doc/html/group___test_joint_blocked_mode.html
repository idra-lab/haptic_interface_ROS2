#include <iostream>
#include <chrono>
#include <thread>
#include <array>
#include "RaptorAPI.hpp"
float RAD2DEG(float &a)
{
return static_cast<float>((a / M_PI) * 180.0);
}
float DEG2RAD(float &a)
{
return static_cast<float>((a * M_PI) / 180.0);
}
int main(int argc, char **argv)
{
if (argc != 3)
{
std::cout << "Syntax: TestCalibration <device_param_file> <protocol_params>" << std::endl;
std::cout << "Example: TestCalibration virtuose_6d_n187.param localip=127.0.0.1:localport=12120:remoteip=192.168.100.53:remoteport=5000" << std::endl;
return 0;
}
/*
* Force-feedback device initialization
*/
HAPTION::RaptorAPI raptorHandle;
uint8_t major, minor;
std::string build;
raptorHandle.GetVersion(major, minor, build);
std::cout << "RaptorAPI version " << (uint32_t)major << "." << (uint32_t)minor << " build " << build << std::endl;
raptorHandle.DefineSpecificFields(argv[2]);
HAPTION::ErrorCode error = raptorHandle.Init(argv[1]);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::Init() failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
return -1;
}
std::string robotName{""};
(void)raptorHandle.GetFullName(robotName);
std::cout << "Successfully connected to device " << robotName << std::endl;
HAPTION::CalibrationStatus status;
raptorHandle.GetCalibrationStatus(status);
if (status != HAPTION::CalibrationStatus::C_CALIBRATED)
{
std::cerr << "Please calibrate the device first!" << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle.Close();
return -1;
}
error = raptorHandle.ActivateForceFeedback(true);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ActivateForceFeedback() failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
(void)raptorHandle.Close();
return -1;
}
error = raptorHandle.StartJointPositionMode();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::StartJointPositionMode() failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
(void)raptorHandle.Close();
return -1;
}
HAPTION::JointVector jointKGains{0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F};
HAPTION::JointVector jointBGains{0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F};
raptorHandle.GetMaxJointGains(jointKGains, jointBGains);
raptorHandle.ChangeJointGains(jointKGains, jointBGains);
HAPTION::JointVector blockJointPose{0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F};
HAPTION::JointVector blockJointSpeed{0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F, 0.0F};
/*
* Periodic loop
*/
auto lastStep = std::chrono::system_clock::now();
do
{
std::chrono::system_clock::time_point timeStamp;
error = raptorHandle.ReadState(timeStamp, 0);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ReadState() failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
break;
}
HAPTION::PowerStatus powerStatus;
raptorHandle.GetPowerStatus(powerStatus);
HAPTION::JointVector jointPose;
(void)raptorHandle.GetJointAngles(jointPose);
HAPTION::JointVector motorCurrents;
raptorHandle.GetMotorCurrents(motorCurrents);
std::cout << motorCurrents.v[0] << " " << motorCurrents.v[1] << " " << motorCurrents.v[2] << " " << motorCurrents.v[3] << " "
<< motorCurrents.v[4] << " " << motorCurrents.v[5] << " " << motorCurrents.v[6] << std::endl;
//std::cout << RAD2DEG(jointPose.v0) << " " << RAD2DEG(jointPose.v1) << " " << RAD2DEG(jointPose.v2) << " " << RAD2DEG(jointPose.v3) << " "
//<< RAD2DEG(jointPose.v4) << " " << RAD2DEG(jointPose.v5) << " " << RAD2DEG(jointPose.v6) << std::endl;
//std::cout << jointPose.v0 << " " << jointPose.v1 << " " << jointPose.v2 << " " << jointPose.v3 << " "
//<< jointPose.v4 << " " << jointPose.v5 << " " << jointPose.v6 << std::endl;
(void)raptorHandle.SetExtRegister(0, 0.0F);
(void)raptorHandle.SetExtRegister(1, 0.96F);
(void)raptorHandle.SetExtRegister(2, 0.0F);
(void)raptorHandle.ChangeBrakeStatus(HAPTION::BrakeStatus::BRAKE_RELEASED);
if (powerStatus != HAPTION::PowerStatus::P_POWER_DISINHIBITED)
{
blockJointPose = jointPose;
}
raptorHandle.SetJointAngles(blockJointPose);
raptorHandle.SetJointSpeeds(blockJointSpeed);
error = raptorHandle.SendSetpoints();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::SendSetpoints() failed with error " << (int)error << std::endl;
break;
}
while (true)
{
auto now = std::chrono::system_clock::now();
auto duration = std::chrono::duration_cast<std::chrono::microseconds>(now - lastStep).count();
if (duration > 1000)
{
lastStep = now;
break;
}
}
bool rightButton = false;
raptorHandle.GetOperatorButton(HAPTION::OperatorButton::B_RIGHT, rightButton);
if (rightButton)
{
break;
}
} while (true);
(void)raptorHandle.Close();
std::cerr << "Exiting now!" << std::endl;
return 0;
}
