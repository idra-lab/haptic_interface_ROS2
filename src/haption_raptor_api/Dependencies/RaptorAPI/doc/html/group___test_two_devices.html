#include <iostream>
#include <chrono>
#include <thread>
#include "RaptorAPI.hpp"
int main(int argc, char **argv)
{
if (argc != 5)
{
std::cout << "Syntax: TestTwoDevices <device1_param_file> <protocol1_params> <device2_param_file> <protocol2_params>" << std::endl;
std::cout << "Example: TestCalibration virtuose_6d_n187.param remoteip=192.168.100.53:remoteport=5000 virtuose_6d_n188.param remoteip=192.168.100.54:remoteport=5000" << std::endl;
return 0;
}
/*
* Force-feedback device initialization
*/
HAPTION::RaptorAPI raptorHandle1;
raptorHandle1.DefineSpecificFields(argv[2]);
HAPTION::ErrorCode error = raptorHandle1.Init(argv[1]);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::Init() for first device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
return -1;
}
std::string robotName{""};
(void)raptorHandle1.GetFullName(robotName);
std::cout << "Successfully connected to first device " << robotName << std::endl;
HAPTION::CalibrationStatus status;
raptorHandle1.GetCalibrationStatus(status);
if (status != HAPTION::CalibrationStatus::C_CALIBRATED)
{
std::cerr << "Please calibrate the device first!" << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
return -1;
}
HAPTION::RaptorAPI raptorHandle2;
raptorHandle2.DefineSpecificFields(argv[4]);
error = raptorHandle2.Init(argv[3]);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::Init() for second device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
return -1;
}
(void)raptorHandle2.GetFullName(robotName);
std::cout << "Successfully connected to second device " << robotName << std::endl;
raptorHandle2.GetCalibrationStatus(status);
if (status != HAPTION::CalibrationStatus::C_CALIBRATED)
{
std::cerr << "Please calibrate the device first!" << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
raptorHandle2.Close();
return -1;
}
error = raptorHandle1.ActivateForceFeedback(true);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ActivateForceFeedback() for first device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
raptorHandle2.Close();
return -1;
}
error = raptorHandle2.ActivateForceFeedback(true);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ActivateForceFeedback() for second device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
raptorHandle2.Close();
return -1;
}
error = raptorHandle1.StartCartesianPositionMode();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::StartCartesianPositionMode() for first device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
raptorHandle2.Close();
return -1;
}
raptorHandle1.ChangeCartesianGains(0.0F, 0.0F, 0.0F, 0.0F);
error = raptorHandle2.StartCartesianPositionMode();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::StartCartesianPositionMode() for second device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
raptorHandle1.Close();
raptorHandle2.Close();
return -1;
}
raptorHandle2.ChangeCartesianGains(0.0F, 0.0F, 0.0F, 0.0F);
/*
* Periodic loop
*/
auto lastStep = std::chrono::system_clock::now();
size_t counter{ 0U };
do
{
std::chrono::system_clock::time_point timeStamp;
error = raptorHandle1.ReadState(timeStamp, 0);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ReadState() for first device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
break;
}
error = raptorHandle2.ReadState(timeStamp, 0);
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::ReadState() for second device failed with error " << (int)error << std::endl;
std::cerr << "Exiting now!" << std::endl;
break;
}
HAPTION::Displacement manipPose1;
HAPTION::CartesianVector manipSpeed1;
(void)raptorHandle1.GetCartesianPose(manipPose1);
(void)raptorHandle1.GetCartesianSpeed(manipSpeed1);
raptorHandle1.SetCartesianPose(manipPose1);
raptorHandle1.SetCartesianSpeed(manipSpeed1);
error = raptorHandle1.SendSetpoints();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::SendSetpoints() for first device failed with error " << (int)error << std::endl;
break;
}
HAPTION::Displacement manipPose2;
HAPTION::CartesianVector manipSpeed2;
(void)raptorHandle2.GetCartesianPose(manipPose2);
(void)raptorHandle2.GetCartesianSpeed(manipSpeed2);
raptorHandle2.SetCartesianPose(manipPose2);
raptorHandle2.SetCartesianSpeed(manipSpeed2);
error = raptorHandle2.SendSetpoints();
if (error != HAPTION::ErrorCode::E_NOERROR)
{
std::cerr << "HAPTION::RaptorAPI::SendSetpoints() for second device failed with error " << (int)error << std::endl;
break;
}
if (counter % 1000 == 0)
{
std::cout << "First device position: "
<< manipPose1.t_x << ", "
<< manipPose1.t_y << ", "
<< manipPose1.t_z << std::endl;
std::cout << "Second device position: "
<< manipPose2.t_x << ", "
<< manipPose2.t_y << ", "
<< manipPose2.t_z << std::endl;
}
counter ++;
while (true)
{
auto now = std::chrono::system_clock::now();
auto duration = std::chrono::duration_cast<std::chrono::microseconds>(now - lastStep).count();
if (duration > 1000)
{
lastStep = now;
break;
}
}
bool rightButton = false;
raptorHandle1.GetOperatorButton(HAPTION::OperatorButton::B_RIGHT, rightButton);
if (rightButton)
{
break;
}
raptorHandle2.GetOperatorButton(HAPTION::OperatorButton::B_RIGHT, rightButton);
if (rightButton)
{
break;
}
} while (true);
(void) raptorHandle1.Close();
(void) raptorHandle2.Close();
std::cerr << "Exiting now!" << std::endl;
return 0;
}
